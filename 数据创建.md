### 第三步：核心代码（LangChain+Chroma构建RAG知识库，转向量+本地存储）
这是核心步骤，用**LangChain**拆分文本、**通义千问Embedding**将文本转成AI可检索的**向量格式**、**Chroma**将向量存入**本地数据库**，生成可直接调用的服装电商RAG知识库，**全程代码实现，不用手动操作**。
#### RAG构建代码（核心，替换API-Key即可）
```python
from langchain.document_loaders import TextLoader
from langchain.text_splitter import CharacterTextSplitter
from langchain.vectorstores import Chroma
from langchain.embeddings.dashscope import DashScopeEmbeddings
import os

# -------------------------- 关键配置项（必须替换！） --------------------------
# 1. 你的通义千问API-Key（申请的dashscope api-key）
DASHSCOPE_API_KEY = "你的通义千问API-Key"
# 2. 纯文本知识库路径（第二步生成的TXT文件）
TXT_PATH = "你的电脑本地文件夹路径/服装电商知识库.txt"
# 3. Chroma本地向量数据库保存路径（提前创建文件夹，比如C:/AI作业/服装向量库/）
CHROMA_DB_PATH = "你的电脑本地向量库路径/"

# -------------------------- 1. 加载纯文本知识库 --------------------------
# 用LangChain的TextLoader加载TXT文件
loader = TextLoader(TXT_PATH, encoding="utf-8")
documents = loader.load()
print(f"✅ 成功加载服装知识库，总字符数：{len(documents[0].page_content)}")

# -------------------------- 2. 拆分文本（LangChain核心，适配向量检索） --------------------------
# 拆分规则：chunk_size=200（每个片段200字符），chunk_overlap=30（片段重叠30字符，避免断句）
text_splitter = CharacterTextSplitter(
    chunk_size=200,
    chunk_overlap=30,
    separator="\n",
    strip_whitespace=True
)
# 拆分后的文本片段
texts = text_splitter.split_documents(documents)
print(f"✅ 文本拆分完成，生成{len(texts)}个检索片段")

# -------------------------- 3. 初始化通义千问Embedding（转向量） --------------------------
# 用通义千问的Embedding模型，免费，中文适配性强
os.environ["DASHSCOPE_API_KEY"] = DASHSCOPE_API_KEY
embeddings = DashScopeEmbeddings(model="text-embedding-v1")  # 通义千问免费嵌入模型

# -------------------------- 4. Chroma存储向量，构建RAG知识库 --------------------------
# 将拆分后的文本转向量，存入Chroma本地数据库
db = Chroma.from_documents(
    documents=texts,
    embedding=embeddings,
    persist_directory=CHROMA_DB_PATH  # 本地保存，下次可直接调用
)
# 持久化数据库（必须执行，否则向量不会保存）
db.persist()
# 验证：检索一次测试
test_query = "160/90斤穿什么码？"
docs = db.similarity_search(test_query, k=1)  # k=1：取最相关的1个片段
print(f"\n🚀 RAG知识库构建完成！本地向量库保存至：{CHROMA_DB_PATH}")
print(f"🔍 测试检索「{test_query}」，匹配结果：{docs[0].page_content}")
```

构建好的RAG知识库可直接调用，代码实现**「用户提问→Chroma检索知识库精准答案→传给大模型→生成贴合服装场景的回复」**，完美对接你的服装电商智能体，这是智能体的核心逻辑。
#### 知识库调用+大模型对接代码
```python
from langchain.vectorstores import Chroma
from langchain.embeddings.dashscope import DashScopeEmbeddings
import dashscope
import os

# -------------------------- 配置项（和第三步一致） --------------------------
DASHSCOPE_API_KEY = "你的通义千问API-Key"
CHROMA_DB_PATH = "你的电脑本地向量库路径/"
dashscope.api_key = DASHSCOPE_API_KEY

# -------------------------- 1. 加载本地Chroma向量库（不用重复构建） --------------------------
embeddings = DashScopeEmbeddings(model="text-embedding-v1")
db = Chroma(persist_directory=CHROMA_DB_PATH, embedding_function=embeddings)
print("✅ 成功加载本地服装RAG知识库！")

# -------------------------- 2. 定义「检索+生成」函数（智能体核心逻辑） --------------------------
def fashion_agent_answer(query):
    # 步骤1：从RAG知识库检索最相关的内容（k=2：取前2个最相关片段）
    docs = db.similarity_search(query, k=2)
    context = "\n".join([doc.page_content for doc in docs])
    # 步骤2：构建提示词（将检索到的服装知识传给大模型）
    prompt = f"""
    你是一名专业的服装电商运营智能体，需根据以下服装行业专属知识回答问题，回答要贴合电商场景，简洁实用：
    【服装专属知识】：{context}
    【用户问题】：{query}
    """
    # 步骤3：调用通义千问大模型生成回复（免费通用模型，不用微调）
    response = dashscope.Generation.call(
        model="qwen-turbo",  # 通义千问免费轻量模型，适合文本生成
        prompt=prompt,
        max_tokens=512,
        temperature=0.7  # 生成多样性，0.7适合电商文案/客服
    )
    # 返回结果
    if response.status_code == 200:
        return response.output["text"]
    else:
        return f"❌ 回答失败：{response.message}"

# -------------------------- 3. 测试智能体功能（服装电商核心场景） --------------------------
if __name__ == "__main__":
    # 测试1：客服问答
    query1 = "160cm/95斤穿什么码？"
    # 测试2：标题生成
    query2 = "生成法式雪纺连衣裙的淘宝标题，嵌入热搜词"
    # 测试3：文案生成
    query3 = "生成梨形身材牛仔裤的小红书种草文案"
    
    print(f"📌 问题1：{query1}")
    print(f"💡 回答1：{fashion_agent_answer(query1)}\n")
    
    print(f"📌 问题2：{query2}")
    print(f"💡 回答2：{fashion_agent_answer(query2)}\n")
    
    print(f"📌 问题3：{query3}")
    print(f"💡 回答3：{fashion_agent_answer(query3)}")
```
